##### Openssl command to create Server certificate #####
openssl req -x509 -nodes -days 1825 -newkey rsa:2048 -keyout example.com.key -out example.com.crt -subj "/CN=*.example.com"


##### Openssl command to create Client certificate #####
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout client1.key -out client1.crt
openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout client2.key -out client2.crt


##### curl commands to test Client certificates #####
curl -k https://www.example.com  --cert ./nginx-plus/etc/ssl/client1.crt --key ./nginx-plus/etc/ssl/client1.key
curl -k https://www.example.com  --cert ./nginx-plus/etc/ssl/client2.crt --key ./nginx-plus/etc/ssl/client2.key


##### Openssl command to encrypt key file with a pass-phrase #####
openssl rsa -aes256 -in example.com.key -out example.com.withpass.key
